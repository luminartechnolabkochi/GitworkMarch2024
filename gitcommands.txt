>git status [list all untracked files]

>git init [initialize git in pwd]

>git add -A [adding all untracked files to statging area]

>git commit -m "commit message" [adding staged files to local repository]

>git push [lR=>RR]

>git pull [RR=>changes=>LWS]

>git clone [RR=>exact copy =>LWS]

>git remote add remote_name remote_path [adding remote]


>git push remote_name branch [push]

>git branch [list all branches]
